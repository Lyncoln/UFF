2 - Integrais

Podemos usar o R para obter respostas numricas de uma integral.

2.1 - Integrao em uma dimenso

Suponha que desejamos calcular a seguinte integral \[\int_{0}^{0.1}5e^{-5x}I(x>0)dx.\]

funcao = function(x){
  valor = 5*exp(-5*x)*I(x>0)
  return(valor)
}

Como obter a rea da integral acima?

integrate(funcao, lower = 0, upper = 0.1)
## 0.3934693 with absolute error < 4.4e-15

Ser que o valor obtido acima est correto? O que sabemos sobre a funo funcao criada?

pexp(0.1, rate = 5)
## [1] 0.3934693
integrate(funcao, lower = 0, upper = Inf)
## 1 with absolute error < 2.3e-06

2 - Derivadas

Podemos tambm realizar operaes de derivao no R. Suponha que desejamos calcular \[\frac{d(e^{-x} cos(x) y +3y^2x)}{dx}.\]

#Primeiro usamos o quote para criar a funo
f = quote(exp(-x)*cos(x)*y + 3*y^2*x)

#Visualizando f
f
## exp(-x) * cos(x) * y + 3 * y^2 * x
#Depois usamos a funao D para derivar
df.dx = D(f,"x")

#Visualizando o resultado da derivao
df.dx
## 3 * y^2 - (exp(-x) * sin(x) + exp(-x) * cos(x)) * y

E se desejarmos obter a derivada segunda?

D(df.dx, "x")
## -((exp(-x) * cos(x) - exp(-x) * sin(x) - (exp(-x) * sin(x) + 
##     exp(-x) * cos(x))) * y)

Se quisssemos calcular \[\frac{\partial^2(e^{-x} cos(x) y +3y^2x)}{\partial x \partial y}.\]

D(df.dx, "y")
## 3 * (2 * y) - (exp(-x) * sin(x) + exp(-x) * cos(x))

3 - Maximizando funes no R

Suponha que estamos interessado em maximizar a funo \[f(x) = 2 - (x - 3)^2, \qquad -3 < x < 5.\]

Podemos incialmente explorar a funo com um grfico.

#Criando a funo
fun.max = function(x){
  valor = 2 - (x - 3)^2
  return(valor)
}

#Plotando a funo
library(ggplot2)

ggplot(data = data.frame(x = c(-3,5)), aes(x = x)) + stat_function(fun = fun.max)

Para obtermos o mximo, basta utilizarmos a funo optimise.

#Obtendo o mximo
maximo = optimise(fun.max,c(-3,5), maximum = TRUE)

#Visualizando o mximo
maximo
## $maximum
## [1] 3
## 
## $objective
## [1] 2
#Apresentando o mximo no grfico
ggplot(data = data.frame(x = c(-3,5)), aes(x = x)) + stat_function(fun = fun.max) + geom_vline(aes(xintercept = maximo$maximum), linetype = "dashed", color = "red" )

Qual o nosso principal interesse em maximizar funes?

Suponha que estamos interessados em encontrar a estimativa de mxima verossimilhana do parmetro \(\lambda\) da distribuio de Poisson, se foi observada uma amostra com os seguintes valores \[z = c(2,3,1,2,3,4,5,3,3,1).\]

Basta seguirmos os passos abaixo:

#Criando a funo
ver = function(l,z){
  valor = exp(-l*length(z))*l^{sum(z)}*prod(1/factorial(z))
}

#Obtendo o grfico da funo de verossimilhana
ggplot(data = data.frame(x = c(0,10)), aes(x = x)) + stat_function(fun = ver, args = list(z = c(2,3,1,2,3,4,5,3,3,1))) + xlab("z")

#Obtendo a estimativa de mxima verossimilhana
EMV = optimise(ver,c(0,100),maximum = TRUE,  z=c(2,3,1,2,3,4,5,3,3,1))

EMV

4 - Verificando resultados tericos por meio de simulao

4.1 - Transformaes de variveis

Seja \(X \sim N(0,1)\) e \(Y \sim \chi_{(10)}\). Qual a distribuio de \[\dfrac{X}{\sqrt{Y/10}}?\]

#Vamos checar esse resultado numericamente. Vamos definir um n
n = 10

#Gere uma amostra de X
normal = rnorm(n)

#Gere uma amostra de Y
qui = rchisq(n,10)

#Obtenha o valor da transformao
base2 = data.frame(t = normal/sqrt(qui/10))

#Vamos visualizar graficamente
ggplot(data = base2, aes(x = t)) + geom_histogram(aes(y = ..density..), bins = 10) 

#Se soubermos a distribuio da transformao, podemos acresent-la nos valores obtidos com a transformao para checar se o resultado se aproxima
ggplot(data = base2, aes(x = t)) + geom_histogram(aes(y = ..density..), bins = 10) + stat_function(fun = dt, args = list(df = 10), col = "red")

Ser que 10 valores suficiente para visualizarmos o resultado numericamente?

4.2 - Teorema Central do Limite

Teorema Central do Limite

Para uma amosra aleatria simples \((X_1,\ldots,X_n)\) retiradas de uma populao com mdia \(\mu\) e varincia \(\sigma^2\) finita, a distribuio amostral da mdia \(\bar{X}\) aproxima-se, para n grande, de uma distribuio normal, com mdia \(\mu\) e varincia \(\sigma^2/n\).

Vamos inicialmente considerar a populao formada pelos elementos \(\{1,3,5,5,7\}\).

#Definindo a nossa populao
x = c(1,3,5,5,7)

#Vamos definir o tamanho da amosra que iremos trabalhar
n = 2

#Vamos definir o nmero de amostras da dstribuio de x_barra
num.amostra = 1000

#Vamos retirar 1000 amostras de tamanho 2 e armazenar em uma matriz
mat.amostras = matrix(NA, ncol=num.amostra, nrow=n)

for(i in 1:num.amostra){
  mat.amostras[,i] = sample(c(1,3,5,5,7),n,replace = TRUE)
}

#Calcular a mdia para todas as amostras obtidas
base3 = data.frame(media.amostral = apply(mat.amostras,2,mean))

#Visualizar o histograma das mdias amsotrais

ggplot(data = base3, aes(x = media.amostral)) + geom_histogram(aes(y = ..density..), bins = 15) 

#Acrescentar o grfico da distribuio terica indicada pelo TCL.
media = mean(x)

media
## [1] 4.2
vari =((5-1)/5 * var(x)) / n

vari
## [1] 2.08
ggplot(data = base3, aes(x = media.amostral)) + geom_histogram(aes(y = ..density..), bins = 15) + stat_function(fun = dnorm, args = list(mean = media, sd = sqrt(vari)), col = "red")

#Acrescentando a mdia amostral (azul) e a mdia populacional(vermelho)
ggplot(data = base3, aes(x = media.amostral)) + geom_histogram(aes(y = ..density..), bins = 15) + stat_function(fun = dnorm, args = list(mean = media, sd = sqrt(vari)), col = "red") + geom_vline(aes(xintercept = mean(media.amostral)), linetype = "dashed", color = "red" ) + geom_vline(aes(xintercept = media), linetype = "dashed", color = "blue" )

O resultado parece no estar muito bom, certo?! Mas o que o TCL nos fala sobre o tamanho da amostra retirada da populao (n deve ser grande).
